354 template<class T>
355 bool within(const T &p) const{
356 if (p.x < minDim.x || p.y < minDim.y || p.z < minDim.z)
357 return false;
358 if (p.x > maxDim.x || p.y > maxDim.y || p.z > maxDim.z)
359 return false;
360 cv::Point3d d(p.x - center.x, p.y - center.y, p.z - center.z);
361 if (fabs(u.dot(d)) > l_2)
362 return false;
363 if (fabs(v.dot(d)) > w_2)
364 return false;
365 if (fabs(w.dot(d)) > h_2)
366 return false;
