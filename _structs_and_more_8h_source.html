160std::pair<cv::Point3d, cv::Point3d> find_min_max_3D(const std::vector<T> &points) {
161 double minX = DBL_MAX, maxX =
162 -DBL_MAX, minY = DBL_MAX, maxY = -DBL_MAX, maxZ =
163 -DBL_MAX;
164 double minZ =DBL_MAX;
165 for (const auto &p : points) {
166 if (p.x > maxX)
167 maxX = p.x;
168 if (p.y > maxY)
169 maxY = p.y;
170 if (p.z > maxZ)
171 maxZ = p.z;
172 if (p.x < minX)
173 minX = p.x;
174 if (p.y < minY)
175 minY = p.y;
176 if (p.z < minZ)
177 minZ = p.z;
178 }
179 return {cv::Point3d(minX,minY,minZ),cv::Point3d(maxX,maxY,maxZ)};
180}
